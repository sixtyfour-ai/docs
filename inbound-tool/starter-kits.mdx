## Integration Examples

### CRM Integration

```javascript
// Salesforce Integration Example
import { InboundTool } from '@sixtyfour/inbound-tool';
import { SalesforceClient } from '@sixtyfour/salesforce-integration';

const inboundTool = new InboundTool({
  apiKey: 'your-api-key'
});

const salesforce = new SalesforceClient({
  clientId: 'your-client-id',
  clientSecret: 'your-client-secret'
});

// Sync leads between systems
async function syncLeads() {
  const leads = await inboundTool.processLeads();
  await salesforce.createLeads(leads);
}
```

### Marketing Automation

```javascript
// HubSpot Integration Example
import { InboundTool } from '@sixtyfour/inbound-tool';
import { HubSpotClient } from '@sixtyfour/hubspot-integration';

const inboundTool = new InboundTool({
  apiKey: 'your-api-key'
});

const hubspot = new HubSpotClient({
  apiKey: 'your-hubspot-key'
});

// Create marketing campaigns
async function createCampaign(leads) {
  const campaign = await hubspot.createCampaign({
    name: 'New Leads Campaign',
    leads: leads
  });
  return campaign;
}
```
